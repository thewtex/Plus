PROJECT(VTKFLOCK)
INCLUDE_REGULAR_EXPRESSION("^flock.*$")

SET (VTKFLOCK_INCLUDE_DIRS ${FLOCK_SOURCE_DIR} ${FLOCK_BINARY_DIR} )

SET(common_SRCS
flock.c)

IF (WIN32)
  IF (NOT BORLAND)
    IF (NOT CYGWIN)
      SET(common_SRCS ${common_SRCS} vtkflock.def)
    ENDIF (NOT CYGWIN)
  ENDIF (NOT BORLAND)
ENDIF (WIN32)

INCLUDE_DIRECTORIES( ${VTKFLOCK_INCLUDE_DIRS} )
SET (Tracking_INCLUDE_DIRS ${Tracking_INCLUDE_DIRS} ${VTKFLOCK_INCLUDE_DIRS} CACHE INTERNAL "")

ADD_LIBRARY(vtkflock ${common_SRCS})
TARGET_LINK_LIBRARIES( vtkflock )

INSTALL_TARGETS(/lib/vtk vtkflock)
